# max penalty in seconds
mesecons_debug.max_penalty (maximum penalty) int 300

# everything above this threshold will disable the mesecons in that mapblock
mesecons_debug.penalty_mapblock_disabled (threshold to disable mapblock) int 60

# time between /mesecons_clear_penalty commands, in seconds
mesecons_debug.penalty_clear_cooldown (penalty clear command cooldown interval) int 120

# cpu usage in microseconds that triggers the penalty mechanism
mesecons_debug.max_usage_micros (usage to trigger penalties) int 15000

# remove unmodified penalty data for a mapblock from memory after this many microseconds
mesecons_debug.cleanup_time_micros (penalty data lifetime) int 300000000

# interval between running penalty checks
mesecons_debug.penalty_check_interval (penalty check interval) int 1
